Our LFBI Staff are devoted to supporting your spiritual development. Please refer to the email addresses below to contact us for help. 

If you are unsure who to contact, you can always file a support ticket [here](https://www.cognitoforms.com/mbt3/lfbihelpdesk), and you will be connected with the appropriate resource.

import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid>
	<Card title="Technical Support" icon="email">
		If you are experiencing a technical issue, contact [tech@lfbi.org](mailto:tech@lfbi.org).
        This includes problems with logging in to MyLFBI, viewing lectures, or submitting assignments.
	</Card>
	<Card title="Admissions" icon="email">
		If you have a question or issue regarding admission or enrollments contact [admissions@lfbi.org](mailto:admissions@lfbi.org).
	</Card>
	<Card title="Academic Affairs" icon="email">
		If you have a question or issue regarding academic affairs, contact [studentaffairs@lfbi.org](mailto:studentaffairs@lfbi.org).
        This includes academic standing (warning and probation statuses) and the mentorship program.
	</Card>
	<Card title="Registrar" icon="email">
		For any other academic matters, contact [registrar@lfbi.org](mailto:registrar@lfbi.org).
        This includes requests for transcripts or credit transfers.
	</Card>
</CardGrid>
